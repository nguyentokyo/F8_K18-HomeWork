select
    c.customer_id,
    c.name,
    o.count_orders,
    o.total_amount
from customers as c
left join (
	select
	　customer_id
	　,count(*) as count_orders
	　,sum(total_amount) as total_amount
	from orders
	group by customer_id
　) as o on
o.customer_id = c.customer_id
where o.total_amount > 1000000

■cach 2
SELECT
    c.customer_id,
    c.name,
    COUNT(o.order_id) AS count_orders,
    SUM(o.total_amount) AS total_amount
FROM customers c
JOIN orders o
    ON c.customer_id = o.customer_id
GROUP BY
    c.customer_id,
    c.name
HAVING SUM(o.total_amount) > 1000000;

Câu １.3
SELECT
    c.name,
    o.count_orders
FROM customers c
LEFT JOIN (
    SELECT
        customer_id,
        COUNT(*) AS count_orders
    FROM orders
    GROUP BY customer_id
) o
ON c.customer_id = o.customer_id
ORDER BY o.count_orders DESC
LIMIT 5;

Câu 2.1
select *
from students
where class in ('CNTT01','CNTT02')
and gender in ('F')
and exists (
　select *
　from scores
　where students.student_id = scores.student_id
　and scores.score < 5
)

    s.subject_name,
    avg(sc.score) as avg_score,
    count(sc.student_id) as total_students
from scores sc
left join subjects s on
s.subject_id = sc.subject_id
group by
    s.subject_id,
    s.subject_name
having avg(sc.score) < 6.5 and count(sc.student_id) >= 10;

Câu 2.3
select
    st.full_name,
    sb.subject_name,
    sc.score
from students st
left join scores sc on
 st.student_id = sc.student_id
left join subjects sb on
 sb.subject_id = sc.subject_id
where sc.score >= 8
order by
    st.student_id,
    sb.subject_id
limit 5 offset 10;

 select
	c.cat_id
	,c.cat_name
	,sum(p.price * s.quantity) as total_sales
 from categories as c
 left join products as p on
 p.cat_id = c.cat_id
 left join sales as s on
 s.product_id = p.product_id
 group by c.cat_id,c.cat_name
 having sum(p.price x s.quantity) > 50000000

Câu 3.2
select
　e.emp_name
　,count(DISTINCT s.product_id) as count_product
　,sum(s.quantity) as total_quantity
from employees as e
left join sales as s
 s.emp_id = e.emp_id
where e.department = 'Sales'
group by e.emp_name
having count(s.product_id) >= 3
